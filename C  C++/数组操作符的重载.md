## 数组访问方式访问字符串对象
```cpp
#include <iostream>
#include <string>

using namespace std;

int main()
{
    string s = "a1b2c3d4e";//数组访问方式访问  字符串  类的对象
    int n = 0;
        
    for(int i = 0; i<s.length(); i++)
    {
        if( isdigit(s[i]) )
        {
            n++;
        }
    }
    
    cout << n << endl;
    
    return 0;
}

```

## 素组访问操作符
- 数组访问符原生意义数组访问和指针运算

a[n]  ==*(a+n) ==*(n+a) == n[a]

```cpp
#include <iostream>
#include <string>

using namespace std;

int main()
{
    int a[5] = {0};
    
    for(int i=0; i<5; i++)
    {
        a[i] = i;
    }
    
    for(int i=0; i<5; i++)
    {
        cout << *(a + i) << endl;    // cout << a[i] << endl;
    }
    
    cout << endl;
    
    for(int i=0; i<5; i++)
    {
        i[a] = i + 10;    //  *(i + a) ==> *(a + i) ==> a[i]  ==>  a[i] = i + 10;   
    }
    
    for(int i=0; i<5; i++)
    {
        cout << *(i + a) << endl;  // cout << a[i] << endl;
    }
    
    return 0;
}

```


## 数组访问操作符（[]）
- 只能通过类的成员函数重载
- 重载能且能使用一个参数
- 可以定义不停参数的多个重载函数

```cpp
#include <iostream>
#include <string>

using namespace std;

class Test
{
	int a[5];
	
public:
	int& operator [](int i)//注意int&
	{
		return a[i];
	}
	int& operator [](const char* s)
	{
		if(s == "1st")
			return a[0];
		else if(s == "2nd")
			return a[1];
		else if(s == "3rd")
			return a[2];
		else if(s == "4th")
			return a[3];
		else if(s == "5th")
			return a[4];
		
		return a[0];
	}
	int length()
	{
		return 5;
	} 	
};

int main()
{
   Test t;
      
   for(int i=0; i<t.length(); i++)
   {
		t[i] = i;
   }
   
   for(int i=0; i<t.length(); i++)
   {
		cout << t[i] << endl;
   }
   
   cout << t["5th"] << endl;
   cout << t["4th"] << endl;
   cout << t["3rd"] << endl;
   cout << t["2nd"] << endl;
   cout << t["1st"] << endl;
   
   
   
   
   
   
    return 0;
}

```
## 小结
- 数组访问符的重载使对象模拟数组行为
- 只能通过类的成员函数重载数组访问符
- 重载函数能且仅能使用一个参数

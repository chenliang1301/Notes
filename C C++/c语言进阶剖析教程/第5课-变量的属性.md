# 第5课-变量的属性

5-1

```c
#include<stdio.h>

int fun_1(void){

	int i = 0;
	i++;
	return i;
}

int fun_2(void){

	static int n = 0;//静态数据区，只访问一次
	n++;
	return n;
}

int main(void){

	auto int i = 0;
	auto char c1 = 'B';
	register char c = 'A';
	static short s = 3;
		
	printf("i=%d--&i=%p\n",i,&i);
	printf("c1=%d--&c1=%p\n",c,&c1);
	//printf("c=%c--&c=%p\n",c,&c);
	printf("s=%d--&s=%p\n",s,&s);
	for(i=0;i<5;i++){
		printf("fun1=%d--fun2=%d\n",fun_1(),fun_2());
	}
	
	return 0;
}
```

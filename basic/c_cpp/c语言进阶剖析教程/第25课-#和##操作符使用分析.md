# 第25课-#和##操作符使用分析

```c
#include <stdio.h>
#define STRING(f,b) (printf("%s call ...\n",#f),f(b))

int square(int a)
{
    return (a * a);
}

int sum(int a)
{
    return (a + a);
}

int main()
{
    int result = 0;
    
    result = STRING(square, 5);
    printf("result = %d\n",result);
    
    result = STRING(sum, 7);    
    printf("result = %d\n",result);
    return 0;
}
```

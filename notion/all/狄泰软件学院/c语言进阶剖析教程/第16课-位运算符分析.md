16-1.c

```c
#include<stdio.h>
int main(){

	printf("%d\n",1<<2);//4
	printf("%d\n",3>>2);//11 00 0
	printf("%d\n",1<<2+3);//1*2^5 32
	printf("%d\n",3<<-1);// 11 01 1	
	return 0;
}
```

16-2.c

```c
#include<stdio.h>

#define swap1(a, b)         \
{							\
	int temp;				\
	temp = a;				\
	a = b;					\
	b = temp;				\
}

#define swap2(a, b)         \
{							\
	a = a + b;				\
	b = a - b;				\
	a = a - b;				\
}

#define swap3(a, b)         \
{							\
	a = a^b;				\
	b = a^b;				\
	a = a^b;				\
}

int main(){

	int a = 1;
	int b = 2;

	swap1(a, b);
	printf("%d,%d\n",a,b);
	swap2(a, b);	
	printf("%d,%d\n",a,b);
	swap3(a, b);	
	printf("%d,%d\n",a,b);
	return 0;
}
```
